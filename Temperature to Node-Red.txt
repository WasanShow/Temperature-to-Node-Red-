 #include <DHT.h>        

  #define DHTPIN 2         
  #define DHTTYPE DHT22     

  DHT dht(DHTPIN, DHTTYPE);   

  void setup () {
    dht.begin();
    Serial.begin(9600);         
  }

  void loop (){

      float H = dht.readHumidity();     
      float T = dht.readTemperature();    

      String dhtData = String(H) + "," + String(T);
      
      Serial.println(dhtData);
      delay(2000); 
  }
